set(TESTS syev heev)
foreach(name ${TESTS})
  set(target_name core_${name})
  add_executable(${target_name} ${name}.cpp)
  set_target_properties(${target_name} PROPERTIES OUTPUT_NAME ${name})
  target_link_libraries(${target_name} rokko gtest_main)
  add_test(${name} ${name})
endforeach(name)
unset(TESTS)
